echo "Creating virtual environment"

python3 -m venv .venv
source .venv/bin/activate

echo "Installing nltk"
pip install -U --disable-pip-version-check nltk > /dev/null

echo "Installing SpaCy"
pip install -U --disable-pip-version-check spacy > /dev/null

echo "Installing SpaCy Generic Model for NER, POS, etc."
python -W -m spacy download en_core_web_md > /dev/null

echo "Running program"
python3 qa.py ./data/inputfile.txt > ./data/ourtest.answers

echo "Done. Removing virtual environment"
rm -rf .venv
